115 inline QList<QObject*> ElementSet::allElements() const
116 {
117  QList<QObject*> elements;
118 
119  elements.append(dataSource());
120  for (auto sampleDecoder : sampleDecoders())
121  elements.append(sampleDecoder);
122  for (auto featureExtractor : featureExtractors())
123  elements.append(featureExtractor);
124  elements.append(classifier());
125  elements.append(action());
126  elements.append(dataSink());
127 
128  return elements;
