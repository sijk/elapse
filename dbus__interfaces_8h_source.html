233  bool checkConnected()
234  {
235  auto reply = d.isAccessible();
236  reply.waitForFinished();
237  if (!reply.isError())
238  return true;
239 
240  qxtLog->error("The root dbus object is not accessible");
241  qxtLog->error("Interface:", d.lastError().message());
242  qxtLog->error("Server:", reply.error().message());
243  return false;
244  }
